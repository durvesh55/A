import random

# Bot details
name = "Jarvis"
weather = "sunny"
mood = "sad"

# Questions list
questions = [
    "1. What is your name?",
    "2. How are you?",
    "3. What is today's weather?",
    "Type 'exit' to stop"
]

# Responses
responses = {
    "what's your name?": [
        f"My name is {name}",
        f"I am {name}",
        f"You can call me {name}"
    ],
    "what's today's weather?": [
        f"It's {weather} today",
        f"The weather is {weather}"
    ],
    "how are you?": [
        f"I am {mood}",
        f"I'm feeling {mood}",
        f"{mood}! what about you?"
    ],
    "default": [
        "Sorry, I don't understand",
        "Please ask one of the listed questions"
    ]
}

# Get response
def get_response(message):
    return random.choice(responses[message])

# Map user input
def check_message(text):
    if "name" in text:
        return "what's your name?"
    elif "weather" in text:
        return "what's today's weather?"
    elif "how are" in text:
        return "how are you?"
    else:
        return "default"

# Start chatbot
print("BOT: Hello! I can answer:\n")
for q in questions:
    print(q)

print("\nStart chatting...\n")

# Chat loop
while True:
    user_input = input("YOU: ").lower()

    if user_input == "exit" or user_input == "stop":
        print("BOT: Goodbye!")
        break

    mapped = check_message(user_input)
    print("BOT:", get_response(mapped))
