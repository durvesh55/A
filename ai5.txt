import random

# Bot details
name = "Jarvis"
weather = "sunny"
mood = "sad"

# List of questions chatbot can answer
questions = [
    "1. What is your name?",
    "2. How are you?",
    "3. What is today's weather?",
    "Type 'exit' to stop the chatbot"
]

# Predefined responses
responses = {
    "what's your name?": [
        f"My name is {name}",
        f"I am {name}",
        f"You can call me {name}"
    ],
    "what's today's weather?": [
        f"It's {weather} today",
        f"The weather is {weather}"
    ],
    "how are you?": [
        f"I am {mood}",
        f"I'm feeling {mood}",
        f"{mood}! what about you?"
    ],
    "": [
        "I didn't understand that",
        "Can you try asking something else?"
    ],
    "default": [
        "Sorry, I don't understand",
        "Please ask one of the listed questions"
    ]
}

# Function to get response
def get_response(message):
    if message in responses:
        return random.choice(responses[message])
    else:
        return random.choice(responses["default"])

# Function to map user input
def check_message(text):
    if "name" in text:
        return "what's your name?"
    elif "weather" in text:
        return "what's today's weather?"
    elif "how are" in text:
        return "how are you?"
    else:
        return ""

# Start chatbot
print(" BOT: Hello! I can answer the following questions:\n")

for q in questions:
    print(q)

print("\nStart chatting...\n")

# Chat loop
while True:
    user_input = input("YOU: ").lower()

    if user_input == "exit" or user_input == "stop":
        print("BOT: Goodbye!")
        break

    mapped_text = check_message(user_input)
    reply = get_response(mapped_text)

    print("BOT:", reply)
